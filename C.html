พื้นฐานภาษา C++ สำหรับ Arduino

เรียนรู้วิธีการเขียนโปรแกรมสั่งงานสมองกลเบื้องต้นด้วยภาษา C++

สวัสดีนักเรียนทุกคน! ในบทนี้เราจะมาเรียนรู้วิธีการสั่งงาน Arduino ผ่านภาษา C++ ซึ่งเป็นรากฐานสำคัญในการสร้างโครงงานอัจฉริยะต่างๆ ครับ

1. โครงสร้างพื้นฐาน (The Skeleton)

โปรแกรมของ Arduino จะมี 2 ฟังก์ชันหลักที่เราขาดไม่ได้เลยคือ:

void setup() {
  // ทำงานครั้งเดียวตอนเปิดเครื่อง (ใช้ตั้งค่า)
}

void loop() {
  // ทำงานวนซ้ำไปเรื่อยๆ (ใช้สั่งงานหลัก)
}

2. ตัวแปรและประเภทข้อมูล (Variables)

เราใช้ตัวแปรเพื่อเก็บข้อมูล เช่น ค่าจากเซนเซอร์:

  • int: เลขจำนวนเต็ม (เช่น 13, 200)
  • float: เลขทศนิยม (เช่น 25.5, 3.14)
  • bool: ค่าจริง/เท็จ (true/false)
  • เพิ่มเติม: การ #define ไม่ใช่การประกาศตัวแปรโดยตรง แต่เป็นการแทนตัวแปรนั้นด้วยค่าคงที่
#define ledpin 13
int speedMotor = 80;
float VCC = 7.4;
bool status = true;

3. การตัดสินใจ (Conditions)

ใช้ if-else เพื่อให้หุ่นยนต์ของเราคิดเองได้!

if (temp > 30) {
  digitalWrite(ledPin, HIGH); // เปิดพัดลม
} else {
  digitalWrite(ledPin, LOW);  // ปิดพัดลม
}

4. การทำซ้ำ (Loops)

ในบางครั้งเราต้องการให้ Arduino ทำงานบางอย่างซ้ำๆ หลายครั้งตามจำนวนที่เราต้องการ:

For Loop: ใช้เมื่อรู้จำนวนครั้งที่แน่นอน

// สั่งให้ไฟกระพริบ 5 ครั้ง
for (int i = 0; i < 5; i++) {
  digitalWrite(ledPin, HIGH);
  delay(500);
  digitalWrite(ledPin, LOW);
  delay(500);
}

While Loop: ใช้เมื่อต้องการทำซ้ำจนกว่าเงื่อนไขจะเป็นเท็จ

while (digitalRead(buttonPin) == LOW) {
  // รอจนกว่าปุ่มจะถูกกด (ถ้ายังไม่กด ให้ค้างอยู่ใน Loop นี้)
}

5. การควบคุมขา I/O (Digital & Analog)

การควบคุมขา I/O คือวิธีที่ Arduino ใช้สื่อสารกับอุปกรณ์ภายนอก เช่น ปุ่มกด เซนเซอร์ LED มอเตอร์ ฯลฯ โดยแบ่งออกเป็นแบบ Digital (เปิด-ปิด) และ Analog (ค่าละเอียด)

Digital I/O (ค่ามีแค่ 0 หรือ 1):
  • pinMode(pin, mode): กำหนดโหมดขา (INPUT / OUTPUT)
  • digitalRead(pin): อ่านค่าว่าเป็น HIGH หรือ LOW
  • digitalWrite(pin, value): สั่งให้จ่ายไฟหรือหยุดจ่ายไฟ
Analog I/O (ค่าต่อเนื่อง):
  • analogRead(pin): อ่านค่าจากเซนเซอร์ (0 - 1023)
  • analogWrite(pin, value): จ่ายไฟแบบ PWM (0 - 255)
// ตัวอย่าง: การเขียนใช้งานคำสั่ง
int sensorValue = analogRead(A0);      // อ่านค่า 0-1023
int outputValue = sensorValue / 4;    // แปลงให้เหลือ 0-255
analogWrite(ledPin, outputValue);     // ปรับความสว่าง LED

6. การสร้าง Function

นี่คือวิธีที่เราย่อฟังก์ชันต่างๆเข้ามาเป็นฟังก์ชันเดียว

โดยเราจะใช้ void ตามด้วยชื่อฟังก์ชัน เช่น motorControl ("ตรงนี้ใส่ paramemters ที่ต้องการใช้") และตามด้วย ปีกกาเปิดปิด {}

// กำหนดขา drive Motor (แก้ตามการต่อจริง)
#define ENA 6   // ENA เอาไว้ควบคุมสัญญาณ PWM ฝั่ง A (ความเร็ว)
#define IN1 5   // IN1 ควบคุมการจ่ายไฟ (HIGH, LOW)
#define IN2 4   // IN1 ควบคุมการจ่ายไฟ (HIGH, LOW)

#define ENB 11  // ENB เอาไว้ควบคุมสัญญาณ PWM ฝั่ง B (ความเร็ว)
#define IN3 10  // IN1 ควบคุมการจ่ายไฟ (HIGH, LOW)
#define IN4 9   // IN1 ควบคุมการจ่ายไฟ (HIGH, LOW)

// ตัวอย่าง: การสร้างฟังก์ชันควบคุมมอเตอร์
void motorControl(int speedL, int speedR){   // ประกาศฟังก์ชัน
    analogWrite (ENA, speedFL);
    digitalWrite (IN1, HIGH);
    digitalWrite (IN2, LOW);

    analogWrite (ENB, speedFR);
    digitalWrite (IN3, HIGH);
    digitalWrite (IN4, LOW);
}

void setup(){
    motorControl (50, 50);  //ให้มอเตอร์ฝั่งซ้ายและขวาเคลื่อนที่ด้วยความเร็ว 50
    delay (1000);  //หน่วงเวลา 1000 ms (1000 ms = 1 sec)
}

void loop(){
    motorControl (50, 50);  //ให้มอเตอร์ฝั่งซ้ายและขวาเคลื่อนที่ด้วยความเร็ว 50
}

🌟 สรุปสั้นๆ สำหรับบทนี้

1. Structure: setup() รันครั้งเดียว, loop() รันซ้ำตลอดเวลา

2. Variables: int จำนวนเต็ม, float ทศนิยม, bool จริง/เท็จ, #define คือค่าคงที่

3. Condition: if-else ใช้ให้บอร์ด "ตัดสินใจ" ตามเงื่อนไข

4. Loop: for ใช้คุมจำนวนครั้ง, while รอจนกว่าเงื่อนไขจะเปลี่ยน

5. Digital: มีแค่ 2 สถานะ (HIGH/LOW = เปิด/ปิด)

6. Analog Read: รับค่าจากโลกความจริงมาเป็นตัวเลข (0-1023)

7. Analog Write: ส่งค่าไปควบคุมความแรงหรือความสว่าง (0-255)

8. Function: สร้างคำสั่งใช้เอง เพื่อลดโค้ดซ้ำและเรียกใช้ซ้ำได้
